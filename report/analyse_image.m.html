 76         if ~any(channel(:))
 77             continue;

101 function image_mask = mask_colors(image)
102     [num_rows, num_cols, ~] = size(image);
103     num_pixels = num_rows * num_cols;
104     image_mask = zeros(num_pixels, 3);
105     rgb = double(reshape(image, num_pixels, 3));
106     rgbN = double(reshape(normalise_rgb(image, 'approximate'), num_pixels, 3));
107     rN_sdev = std(rgbN(:,1));
108     gN_sdev = std(rgbN(:,2));
109     bN_sdev = std(rgbN(:,3));
110     rN_mean = mean(rgbN(:,1));
111     gN_mean = mean(rgbN(:,2));
112     bN_mean = mean(rgbN(:,3));
113     hsv = reshape(rgb2hsv(image), num_pixels, 3);
114     for c = 1 : num_pixels
115         rN = rgbN(c,1);
116         gN = rgbN(c,2);
117         bN = rgbN(c,3);
118         hue = hsv(c,1) * 360;
119         % current pixel is red
120         if      (hue >= 330 || hue <= 30) && ...
121                 (normal_prob(rN, rN_mean, rN_sdev) <  0.001)
122                     image_mask(c,1) = 1;
123         % current pixel is green
124         elseif  (hue >= 80 && hue < 180) && ...
125                 (normal_prob(gN, gN_mean, gN_sdev) <  0.007)
126                    image_mask(c,2) = 1;
127         % current pixel is blue
128         elseif  (hue >= 150 && hue <= 270) && ...
129                 (normal_prob(bN, bN_mean, bN_sdev) < 0.0000085)
130                    image_mask(c,3) = 1;
131         end
132     end
133     image_mask = reshape(image_mask, num_rows, num_cols, 3);
134     image_mask = remove_noise(image_mask);
135     image_mask = remove_outliers(image_mask);
136     image_mask = enforce_similar_channel_areas(image_mask);
137 end

140 function x = normal_prob(val, mu, sigma)
141     x = 1.0 / (sigma * sqrt(2 * pi)) * exp(-(val - mu) ^ 2 / (2 * sigma ^ 2));
142 end

156 % finds the connected components in each channel of |image| and removes those
157 % that are far away from the centroid of the pixels in that channel
158 % here, 'far away' means more distant than |distance_proprtion_threshold| times
159 % the average distance of each connected component to the channel centroid
160 % this removes big areas of noise such as the big green blob inside of the black
161 % arrow of the red robot in data/1/00000006.jpg
162 function image = remove_outliers(image, distance_proportion_threshold)
163     if nargin < 2
164         distance_proportion_threshold = 0.5;
165     end
166     [~, ~, num_channels] = size(image);
167     for c = 1 : num_channels
168         channel = image(:,:,c);
169         if ~any(channel(:))
170             continue;
171         end
172         channel_properties = regionprops(channel, 'Centroid');
173         channel_centroid = channel_properties.Centroid;
174         regions = bwconncomp(channel);
175         regions_properties = regionprops(regions, 'Centroid', 'PixelIdxList');
176         regions_centroids = {regions_properties.Centroid};
177         distances = cellfun(@(x) norm(x - channel_centroid), regions_centroids);
178         mean_distance = mean(distances);
179         for d = 1 : length(distances)
180             if distances(d) > mean_distance * distance_proportion_threshold
181                 idx = regions_properties(d).PixelIdxList;
182                 channel(idx) = 0;
183             end
184         end
185         image(:,:,c) = channel;
186     end
187 end

190 % we know that the robots are all about the same size - we can thus remove any
191 % channels in the mask that have a much smaller area than the other channels
192 % this catches some problems like the shadow of the blue robot in
193 % data/1/00000095.jpg being detected as a red blob
194 function image = enforce_similar_channel_areas(image, area_proportion_threshold)
195     if nargin < 2
196         area_proportion_threshold = 0.5;
197     end
198     [~, ~, num_channels] = size(image);
199     areas = zeros(num_channels, 1);
200     for c = 1 : num_channels
201         channel = image(:,:,c);
202         if ~any(channel(:))
203             continue;
204         end
205         region_props = regionprops(channel, 'Area');
206         areas(c) = region_props.Area;
207     end
208     max_area = max(areas(areas > 0));
209     for c = 1 : num_channels
210         area = areas(c);
211         if area == 0
212             continue;
213         end
214         if  (area < max_area * area_proportion_threshold) || ...
215             (area > max_area / area_proportion_threshold)
216                 image(:,:,c) = image(:,:,c) * 0;
217         end
218     end
219 end

222 % this filters out all conneceted regions but the biggest one
223 function mask = filter_mask(mask)
224     [x, y, num_channels] = size(mask);
225     for c = 1 : num_channels
226         channel = zeros(x, y);
227         channel = reshape(channel, x * y, 1);
228         blob_info = bwconncomp(mask(:,:,c));
229         blob_list = blob_info.PixelIdxList;
230         [nrows, ncols] = cellfun(@size, blob_list);
231         largest_blob_pixels = blob_list{find(nrows == max(nrows))};
232         for i = 1 : max(nrows)
233             channel(largest_blob_pixels(i)) = 1;
234         end
235         channel = reshape(channel, x, y);
236         mask(:,:,c) = channel;
237     end
238 end

241 % normalises the values of the red, green, and blue channels of |image| in order
242 % to eliminate illumination differences in the image
243 % formula used: {r, g, b} = {r, g, b} / sqrt(r^2 + g^2 + b^2)
244 % if 'approximate' is passed as an additional parameter, instead use
245 % {r, g, b} = {r, g, b} / (r + g + b)
246 % this is approximately two times faster than the exact normalisation
247 function normalised_image = normalise_rgb(image, varargin)
248     approximate = ~isempty(find(strcmpi(varargin, 'approximate')));
249     red = double(image(:,:,1));
250     green = double(image(:,:,2));
251     blue = double(image(:,:,3));
252     if approximate
253         euclid_rgb = red(:,:) + green(:,:) + blue(:,:);
254     else
255         euclid_rgb = sqrt(red(:,:).^2 + green(:,:).^2 + blue(:,:).^2);
256     end
257     red_norm = round(red(:,:) ./ euclid_rgb .* 255);
258     green_norm = round(green(:,:) ./ euclid_rgb .* 255);
259     blue_norm = round(blue(:,:) ./ euclid_rgb .* 255);
260     % some pixels are absolute black (r = g = b = 0) which causes division by
261     % zero errors during normalisation and NaN values in the normalised channels
262     % need to filter these values out
263     red_norm(isnan(red_norm)) = 0;
264     green_norm(isnan(green_norm)) = 0;
265     blue_norm(isnan(blue_norm)) = 0;
266     red_norm = uint8(red_norm);
267     green_norm = uint8(green_norm);
268     blue_norm = uint8(blue_norm);
269     normalised_image = cat(3, red_norm, green_norm, blue_norm);
270 end
